#include <stdio.h>
int main() {
    int i,n,x,count=0;
    scanf("%d",&n);
    int marks[n];
    for(i=0;i<n;i++) {
        scanf("%d",&marks[i]);
    }
    scanf("%d",&x);
    for(i=0;i<n;i++) {
        if(marks[i]==x) {
            count++;
        }
    }
    printf("%d",count);
    return 0;
}
